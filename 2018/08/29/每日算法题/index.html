思路：刚开始想的是循环去去排序去重然后就取到了第三个大的数字么，但是这个的时间复杂度太高，应该一次遍历解决战斗，这样的话我就是用三个值去做标识，遍历一边去交换这三个tag然后去找出第三个最大的数字.

/*
 * 第三大的数
 * case0: 时间复杂度是O(n)
 * case1：没有三个数 返回最大的数（2个元素或者一个元素）（三个数一样的情况）
 * case2：有三个数，排序返回第三个数
 * case3: 数组中有相等的元素
 * */
int thirdMax(int* nums, int numsSize) {
    int i = 0,count =0;
    long min,mid,max;
    mid = min = max = INTMAX_MIN;

    for (i = 0; i < numsSize ; ++i) {
//        排除相同的项 最后一个相同的话无所谓
        if (nums[i] == max || nums[i] == mid)
        {
            continue;
        }
//        大于最大的
        if (nums[i]>max)
        {
            min = mid;
            mid = max;
            max = nums[i];
            count ++ ; //改变的次数 只有大于三次 证明存在第三个大的数字
        }
//        大于中间的
        else if (nums[i] > mid)
        {
            min = mid;
            mid = nums[i];
            count ++;
        } else if (nums[i] >= min) //小于mid 大于等于 min
        {
            min = nums[i];
            count ++;
        }
    }
    if (count >=3)
    {
//        第三个大的数字
        return min;
    }
    else
    {
//       返回最大的数字
        return max;
    }
}
