排序算法学习

排序算法

在做leetcode的题目“拆分数组”的题目遇到了排序的问题，所以来总结学习下。

冒泡排序

for(int i = 0; i<numsSize;i++)
		{
			for(int j=0; j<numsSize-1-i;j++)
			{
				if(nums[j]>nums[j+1])
				{
					int temp = nums[j+1];
					nums[j+1] = nums[j];
					nums[j] = temp;
				}
			}
		}

进阶版冒泡排序法

int sum,j, left = 0, right = numsSize - 1;
while (left < right) {
	for (j = left; j < right; j++)
		if (nums[j] > nums[j + 1])
           {
               int temp = nums[j+1];
			nums[j+1] = nums[j];
			nums[j] = temp;
           }
	right--;
	for (j = right; j > left; j--)
		if (nums[j - 1] > nums[j])
           {
			int temp = nums[j];
			nums[j] = nums[j-1];
			nums[j-1] = temp;
           }
	left++;
   }

选择排序

int i ,j;
		int sum = 0;
		for (i=0;i<numsSize-1;i++)
		{
			int k =i; 最小值的下标 开始假设为0
			for (j=i+1;j<numsSize;j++)
			{
				if (nums[k]>nums[j])
				{
					k=j; //交换下标
				}
			}
			if (k!=i)
			{
				int temp = nums[i];
				nums[i] = nums[k];
				nums[k] = temp;
			}
		}

希尔排序

 int sum =0;
    int i,j,gap;
    for (gap = numsSize/2;gap>0;gap/=2)  //步长
    {
        for (i=0;i<gap;i++)  //直接插入排序 根据步长分为几组
        {
            for (j=i+gap;j<numsSize;j+=gap) //根据步长取元素 j为元素下标
            {
//                判断是否前后大小颠倒
                if (nums[j]<nums[j-gap]) {
                    int temp = nums[j];
                    int k =j-gap;
                    while (k>=0&&nums[k]>temp)
                    {
                        nums[k+gap] = nums[k];
                        k-=gap;
                    }
                    nums[k+gap] = temp;
                }
            }
        }

    }
